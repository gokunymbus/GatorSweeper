export function getRandomInt(min, max) {
    min = Math.ceil(min);
    max = Math.floor(max);
    return Math.floor(Math.random() * (max - min) + min); //The maximum is exclusive and the minimum is inclusive
}
 
export function tileData(row, column) {
    const isMeow = getRandomInt(1, 7) == 1 ? true : false;
    return {
        // proximities: 0, To be added later
        isMeow: isMeow,
        isRevealed: false,
        row,
        column
    }
}
 
export function generateGrid(rowLength, columnLength, getTileData = (row, column) => {}) {
 
    function buildColumns(currentColumnLength, colArray, currentRowIndex, currentColumnIndex = 0) {
        if (currentColumnLength == 0) {
            return colArray;
        }
 
        const columnArray = colArray.concat(
            getTileData(
                currentRowIndex,
                currentColumnIndex
            )
        );
 
        const newColumnLength = currentColumnLength - 1;
        const newRowIndex = currentRowIndex;
        const newColumnIndex = columnArray.length;
        return buildColumns(newColumnLength, columnArray, newRowIndex, newColumnIndex);
    }
 
    function buildRows(currentRowLength, currentArray, currentRowIndex = 0) {
        if (currentRowLength == 0) {
            return currentArray;
        }
 
        // @TODO Still could be some nested objects that are not being cloned.
        const newRowsArray = [...currentArray, buildColumns(columnLength, [], currentRowIndex)];
        const newRowLength = currentRowLength - 1; 
        return buildRows(newRowLength, newRowsArray, newRowsArray.length);
    }
    
   return buildRows(rowLength, []);
};
 
/**
 * A recursive function that given a target row, column returns the total
 * number of meows in proximety to that target.
 * 
 * @param {*} params Object
 * @returns int The total number of tiles in the perimeter of the target
 */
export function processPerimeter(params) {
    const {
        previousTotal = 0,
        currentPerimeterLength,
        currentColumn,
        currentRow,
        startingRow,
        startingColumn,
        origArray,
        perimeterSize
    } = params;
 
    if (currentPerimeterLength < 0) {
        return previousTotal;
    }
 
    const foundValue = origArray[currentRow] && origArray[currentRow][currentColumn];
    const newTotal = (foundValue && foundValue.isMeow) ? previousTotal + 1 : previousTotal;
    const decrementColumn = currentColumn - 1;
    let newColumn = decrementColumn;
    let newRow = currentRow;
    if (currentPerimeterLength % perimeterSize  == 0) {
        newRow = currentRow - 1;
        newColumn = startingColumn;
    }
 
    const newPerimLength = currentPerimeterLength -1;
    return processPerimeter(
        {
            previousTotal: newTotal,
            currentPerimeterLength: newPerimLength,
            currentColumn: newColumn,
            currentRow: newRow,
            startingRow,
            startingColumn,
            origArray,
            perimeterSize
        }
    );
}
 
export function addProximities(gridArray, perimeterSize = 3) {
 
    const proximitesArray = gridArray.map((row, rowIndex, origArray) => {
        return row.map((column, columnIndex) => {
            if (column.isMeow) {
                return {...column};
            }
            const offset = Math.floor(perimeterSize/2);
            const meowsInPerimeter = processPerimeter(
                {
                    previousTotal: 0,
                    currentPerimeterLength: perimeterSize * perimeterSize,
                    currentColumn: columnIndex + offset,
                    currentRow: rowIndex + offset,
                    startingRow: columnIndex + offset,
                    startingColumn: rowIndex + offset,
                    origArray,
                    perimeterSize
                }
            );
            
            console.log(meowsInPerimeter, 'Target proximities');
            return {...column, "proximities": meowsInPerimeter};
        });
    });
 
    console.log(JSON.stringify(proximitesArray));
    
}
